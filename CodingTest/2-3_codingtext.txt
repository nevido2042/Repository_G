#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> dots) 
{
    int answer = 0;
    
    int minX = 256;
    int maxX = -256;
    int minY = 256;
    int maxY = -256;
    
    for(const vector<int> v : dots)
    {
        minX = min(minX,v[0]);
        maxX = max(maxX,v[0]);
        
        minY = min(minY,v[1]);
        maxY = max(maxY,v[1]);
    }
    
    answer = (maxX - minX) * (maxY - minY);
    
    
    return answer;
}
