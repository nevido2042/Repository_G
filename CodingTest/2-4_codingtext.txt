#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> arr, vector<bool> flag) 
{
    vector<int> answer;
    
    for(int i = 0; i < flag.size(); ++i)
    {
        if(flag[i] == true)
        {
            int cnt = arr[i] * 2;
            
            for(int j = 0; j < cnt; ++j)
            {
                answer.push_back(arr[i]);
            }
        }
        else
        {
            int cnt = arr[i];
            for(int j = 0; j < cnt; ++j)
            {
                answer.pop_back();
            }
        }
    }
    
    return answer;
}
